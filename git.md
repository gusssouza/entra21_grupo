# Comandos básicos de Git

````bash
git config --global user.name "Nome"

git config --global user.email "user@gmail.com"


cd # abrir uma pasta
ls # para ver o que tem na pasta
mkdir # criar pasta
mv # para mudar nome 
clear # limpar
git init # monitorar
git status # status
+git add 'arquivo' # add a arquivo
git rm --cached # desfazer
+git commit -m "Detalhes" # commitar
git branch -M main # mudar de master para main
git remote add origin 'LINK'  # conectar
git remote -vv # mostrar a conecção
+git push origin main # para mandar para o GitHub

git clone 'LINK' # clona o repositório do link
git checkout -b "branch" # nova branch
git checkout  'branch' # ir para um branch
git merge 'branch' # juntar branch na main
git pull origin main # pegar code do Git

````
